%{
#include <stdio.h>

int icount = 0;
int factcount = 0;
%}

%%
fact        { factcount++; }
i           { icount++; }
(.|\n)      ;
%%

int yywrap(void)
{
    return 1;
}

int main()
{
    yylex();
    printf("Count of \"fact\" = %d\nCount of letter 'i' = %d\n",
           factcount, icount);
    return 0;
}
